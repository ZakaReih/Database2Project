CREATE OR REPLACE PROCEDURE group_customers_by_age
IS
  -- Declare variables to store the grouped data
  v_age Customers.age%TYPE;
  v_customer_count NUMBER;
  -- Define a cursor to select the data from the table
  CURSOR c_customers IS 
    SELECT age, COUNT(*) AS customer_count
    FROM Customers
    GROUP BY age;
BEGIN
  -- Open the cursor
  OPEN c_customers;
  -- Loop through the cursor and print out the grouped data
  LOOP
    FETCH c_customers INTO v_age, v_customer_count;
    EXIT WHEN c_customers%NOTFOUND;
    -- Print out the grouped data
    DBMS_OUTPUT.PUT_LINE('Age: '|| v_age ||', Customer Count: ' || v_customer_count);
  END LOOP;
  -- Close the cursor
  CLOSE c_customers;
END;


BEGIN
  group_customers_by_age;
END;
